CREATE DATABASE libroBD;
USE libroBD;

CREATE TABLE tb_autor(
    id INT PRIMARY KEY AUTO_INCREMENT NOT NULL,
    nombre VARCHAR(200),
    nacionalidad VARCHAR(200)
);

/* Tabla para los libros incluye uni√≥n de tablas y establece NULL al eliminar autor */
CREATE TABLE tb_libro(
    id INT PRIMARY KEY AUTO_INCREMENT NOT NULL,
    titulo VARCHAR(200),
    genero VARCHAR(255),
    numero_paginas INT,
    precio DOUBLE,
    imagen LONGBLOB,
    id_autor INT,
    FOREIGN KEY (id_autor) REFERENCES tb_autor (id) ON DELETE SET NULL
);

/*Tabla para usuarios*/

CREATE TABLE tb_usuario (
	id INT PRIMARY KEY AUTO_INCREMENT NOT NULL,
    username VARCHAR(50),
    pass VARCHAR(50)
);
